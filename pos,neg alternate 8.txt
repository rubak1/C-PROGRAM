
#include<stdio.h>

int main()
{
        int n;
        scanf("%d",&n);
        int a[n],pos[n],neg[n];
        int p=0,q=0;
        for (int i=0;i<n;i++){
               scanf("%d",&a[i]);
               if(a[i]>=0){
                      pos[p++]=a[i];
               } else {
                      neg[q++]=a[i];
               }
        }
        int i=0,k=0,j=0;
        while(i<p && j<q){
               a[k++]=neg[j++];
               a[k++]=pos[i++];
        }
        while(i<p){
               a[k++]=pos[i++];
        }
        while(j<q){
               a[k++]=neg[j++];
        }
        
       
        for(int i=0;i<n;i++){
               printf("%d ",a[i]);
        }
}
