#include<stdio.h>
int main()
{
       int n;
       scanf("%d",&n);
       int a[n];
       for (int i=0;i<n;i++){
              scanf("%d",&a[i]);
       }
       int cs=a[0];
       int ms=a[0];
       for(int i=0;i<n;i++){
              if(a[i]>cs+a[i]){
                     cs=a[i];
              }
              else {
                     cs+=a[i];
              }
              if(cs>ms){
                     ms=cs;
              }
       }
       printf("%d",ms);
}
