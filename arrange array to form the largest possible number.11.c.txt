#include<stdio.h>
#include<string.h>

int main ()
{
       int n;
       scanf("%d",&n);
       char a[n][20];
       for(int i=0;i<n;i++){
              scanf("%s",&a[i]);
       }
        char ab[20],ba[20],temp[20];
       for(int i=0;i<n;i++){
              for(int j=i+1;j<n;j++){
                     strcpy(ab,a[i]);
                     strcpy(ab,a[j]);
                     strcpy(ba,a[i]);
                     strcpy(ba,a[j]);
                     if(strcmp(ab,ba)<0){
                            strcpy(temp,a[i]);
                            strcpy(a[i],a[j]);
                            strcpy(a[j],temp);
                     }
              }
       }
       for (int i=0;i<n;i++){
              printf("%s",a[i]);
       }
}
