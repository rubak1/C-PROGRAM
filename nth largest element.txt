#include <stdio.h>

int main(){
    
    int n;
    scanf("%d",&n);
    int a[n];
    for (int i=0;i<n;i++){
     scanf("%d",&a[i]);
        }
        
        
        int k;                  //decending short 
        scanf("%d",&k);
        for(int i=0;i<=n;i++){
            for(int j=i+1;j<n;j++){
                    if (a[i]<a[j]){
                        int temp = a[i];
                          a[i]= a[j];
                           a[j]=temp;     
                    }
      
        }
    }
    
    if(k==1){
            printf("%d",a[0]);
            return 0;
    }
    int count =1;              //duplicate
    int found =0;
    for(int i=0;i<n;i++){
            if(a[i]!=a[i-1]){
                    count++;
                    if (count == k){
                            printf("%d",a[i]);
                            found=1;
                            break; 
                    }
            }
    }
    if(!found){
            printf("no nth element");
    }
}
