#include <stdio.h>

int main()
{
    int n;
    scanf("%d",&n);
    int a[n];
    int k=0;
    for(int i=0;i<n;i++)
    {
        scanf("%d",&a[i]);
    }
    for(int i=0;i<n;i++)
    {
        if(a[i] !=0)
        {
            a[k++]=a[i];
        }
    }
    while(k<n)
    {
        a[k++]=0;
    }
    for(int i=0;i<n;i++)
    {
        printf("%d ",a[i]);
    }
    return 0;
}
