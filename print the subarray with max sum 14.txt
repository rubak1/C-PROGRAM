#include<stdio.h>
int main()
{
       int n;
       scanf("%d",&n);
       int a[n];
       for (int i=0;i<n;i++){
              scanf("%d",&a[i]);
       }
       int cs=a[0];
       int ms=a[0];
       int start =0,end=0,k=0;
       for(int i=0;i<n;i++){
              if(a[i]>cs+a[i]){
                     cs=a[i];
                     k=i;
              }
              else {
                     cs+=a[i];
              }
              if(cs>ms){
                     ms=cs;
                     start = k;
                     end = i;
              }
       }
       printf("%d\n",ms);
       for(int i=start;i<=end;i++){
              printf("%d ",a[i]);
       }
}
